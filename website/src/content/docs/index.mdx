import { Card, CardGrid } from '@astrojs/starlight/components';

<CardGrid>
  <Card title="Durable Object Runtime" icon="cloud">
    Each actor is a Cloudflare Durable Object — single-threaded, globally addressable, with built-in storage. No servers to manage.
  </Card>
  <Card title="Real-time Sync" icon="rocket">
    WebSocket connections with JSON Patch diffs and checksum deduplication. Clients stay in sync with minimal bytes over the wire.
  </Card>
  <Card title="Type Safety" icon="approve-check">
    Generics enforce event/context alignment at compile time. Zod validates at runtime boundaries. No `any`, no `as` casting.
  </Card>
  <Card title="Public / Private Context" icon="setting">
    Every actor has public state shared with all clients and private state scoped per caller. Built-in data isolation.
  </Card>
  <Card title="XState v5" icon="puzzle">
    Full state machine power — parallel states, guards, delayed transitions, spawned actors. Visualize with Stately Studio.
  </Card>
  <Card title="Framework Integrations" icon="laptop">
    First-class React hooks. SSR support for Next.js and TanStack Start. Framework-agnostic core.
  </Card>
</CardGrid>
